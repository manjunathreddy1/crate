import * as React from "react";
import { Dialog, DialogType, PrimaryButton } from "@fluentui/react";

export const FluentDialogExample: React.FC = () => {
  const dialogContentProps = {
    type: DialogType.normal,
    title: "Dialog Title",
    subText: "This is the dialog content. It appears over a dimmed overlay.",
  };

  const modalProps = {
    isBlocking: false, // Set to true to prevent dismiss by clicking outside
    styles: { main: { maxWidth: 450 } },
  };

  return (
    <>
      <PrimaryButton text="Open Dialog" />
      <Dialog
        hidden={false}
        dialogContentProps={dialogContentProps}
        modalProps={modalProps}
      ></Dialog>
    </>
  );
};
