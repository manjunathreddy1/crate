const getItemTitles = async (context: WebPartContext): Promise<string[]> => {
  const sp = spfi().using(SPFx(context));

  try {
    const items = await sp.web.lists.getByTitle("Your List Name").items.select("Title")();

    // Push each title into a string array
    const titles: string[] = items.map(item => item.Title);
    return titles;

  } catch (error) {
    console.error("Error fetching list items:", error);
    return [];
  }
};
